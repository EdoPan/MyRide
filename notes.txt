CREATE TABLE users (
    username varchar(255),
    password varchar(255),
    isMaintainer boolean
);

CREATE TABLE bikes (
    id INTEGER PRIMARY KEY,
    type varchar(255),
  	price double,
    condition varchar(255),
  	stationID int
);

CREATE TABLE rides (
    id INTEGER PRIMARY KEY,
    startTime date,
  	endTime date,
    bikeID int,
  	username varchar(255)
);

CREATE TABLE stations (
    id INTEGER PRIMARY KEY,
    address varchar(255)
);
